=>SELECT DATA BY USING WHERE CLAUSE
NOTE :: WHERE clause is not only used in SELECT record,it is also used in UPDATE,DELETE....

METHOD 1 Syntax

SELECT * FROM tablename
WHERE conditon;




METHOD 2 Syntax

SELECT column1,column2,column3,...
FROM tablename
WHERE condition;


Operator in WHERE clause
=         equal
>         greater than
<         less than
>=        greater than or equal
<=        less than or equal
<>        greater than or less than or equal ..NOTE :: not equal !=
BETWEEN   
LIKE      
IN
